1. Contare quanti iscritti ci sono stati ogni anno

SELECT COUNT(`id`) AS 'numer of enrolments' , YEAR(`enrolment_date`) AS 'years of enrolments'
FROM `students`
GROUP BY YEAR(`enrolment_date`);



2. Contare gli insegnanti che hanno l'ufficio nello stesso edificio

SELECT COUNT(`id`) AS `teacher_numbers` ,`office_address` 
FROM `teachers` 
GROUP BY `office_address`;


3. Calcolare la media dei voti di ogni appello d'esame

SELECT AVG(`vote`) AS `average of votes` , `exam_id` 
FROM `exam_student` 
GROUP BY `exam_id`;
